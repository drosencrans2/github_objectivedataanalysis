def standardize(DATA):
    '''
    Standardize data
    To be Guassian: mean=0 and std=1
    '''
    mean = np.mean(DATA)
    std = np.std(DATA)
    DATA_norm = (DATA-mean)/std
    return(DATA_norm)

def calc_z_score(population,sample):
    '''
    - Barnes Eq. 93
    - Use for populations > 30
    - Calculate the z-score from a population and a sample of length N 
    '''
    N = len(sample)
    sample_mean = np.mean(sample)
    sample_std = np.std(sample)
    population_mean = np.mean(population)
    population_std  = np.std(population)
    xstd = population_std/np.sqrt(N)
    xstd = sample_std/np.sqrt(N)
    z = np.round((sample_mean-population_mean)/xstd,2)
    prob1 = np.round((1-stats.norm.cdf(np.abs(z)))*100,2)   # one sided (prior knowledge)
    prob2 = np.round((1-stats.norm.cdf(np.abs(z)))*2*100,2) # two-sided test
    return(z,prob1,prob2)

def calc_t_score(population,sample):
    '''
    - Barnes Eq. 96
    - Use for population < 30
    - Calculate the t-score from a population and a sample of length N
    - If pvalue < alpha - reject null hypothesis.
    '''
    N=len(sample)
    sample_mean = np.mean(sample)
    sample_std  = np.std(sample)
    population_mean=np.mean(population)
    t = np.round((sample_mean-population_mean)/(sample_std/(np.sqrt(N-1))),2)
    prob1 = np.round((1-stats.t.cdf(t,N-1))*100,2)   # one sided (prior knowledge)
    prob2 = np.round((1-stats.t.cdf(t,N-1))*2*100,2) # two sided
    pval =stats.ttest_ind(sample,population,equal_var=False)[1]
    return(t,prob1,prob2,pval)

def calc_z_score_2_samples(sample1,sample2,population1,population2):
    '''
    - Barnes Eq. 103
    - standard deviations of populations don't have to equal here
    - i.e., they'll likely be the same population a lot of the time with 
      two time periods (sample1 and sample2) taken out of it
    - The population(s) the samples come from must be Normally distributed
    '''
    N1 = len(sample1)
    N2 = len(sample2)
    sample_mean1 = np.mean(sample1)
    sample_mean2 = np.mean(sample2)
    pop_std1 = np.std(population1)
    pop_std2 = np.std(population2)
    z = (sample_mean1-sample_mean2-0)/np.sqrt((pop_std1**2/N1)+(pop_std2**2/N2))
    return(z)
    
def calc_t_score_2_samples(sample1,sample2,population1,population2):
    '''
    - Barnes Eq. 101,102
    - 
    - The population(s) the samples come from must be Normally distributed
    - The population standard deviations are assumed to equal (and will be 
      if samples come from the same population)
    '''
    N1 = len(sample1)
    N2 = len(sample2)
    sample_mean1 = np.mean(sample1)
    sample_mean2 = np.mean(sample2)
    sample_std1 = np.std(sample1)
    sample_std2 = np.std(sample2)
    pop_std1 = np.std(population1)
    pop_std2 = np.std(population2)
    if pop_std1 != pop_std2:
        print('ATTENTION: pop1 and pop2 standard deviations are not equal.  Use Welsch t-test')
    s = np.sqrt((N1*sample_std1**2+N2*sample_std2**2)/(N1+N2-2)) # this is sigma_hat in Barnes 107
    t = np.round((sample_mean1-sample_mean2-0)/(s*np.sqrt(1/N1+1/N2)),2)
    prob1 = (1-stats.t.cdf(t,N-1))*100   # one sided (a priori)
    prob2 = (1-stats.t.cdf(t,N-1))*2*100 # two sided
    return(t,prob1,prob2)

def confidence_z(sample,level):
    '''
    Calculate the confidence level using a z-stastic
    '''
    lev = 1-(1-(level/100))/2 
    zstat = stats.norm.ppf(lev)
    sample_mean = np.mean(sample)
    sample_std  = np.std(sample)
    lo_limit = sample_mean - zstat * (sample_std/np.sqrt(N))
    hi_limit = sample_mean + zstat * (sample_std/np.sqrt(N))
    return(lo_limit,hi_limit)
    
def confidence_t(sample,level,N):
    '''
    Calculate the confidence interval using a t-statistic
    '''
    lev = 1-(1-(level/100))/2 
    df  = N-1
    tstat = stats.t.ppf(lev,df)
    sample_mean = np.mean(sample)
    sample_std  = np.std(sample)
    lo_limit = sample_mean - tstat * (sample_std/np.sqrt(N-1))
    hi_limit = sample_mean + tstat * (sample_std/np.sqrt(N-1))
    return(lo_limit,hi_limit)

def calc_f_score(sample1,sampl2,population1,population2):
    '''
    - assess the ratio between two sample standard deviations from samples 1 and 2
    - Both N1 and N2 should be small here
    '''
    sample_std1 = np.std(sample1)
    sample_std2 = np.std(sampl2)
    pop_std1 = np.std(population1)
    pop_std2 = np.std(population2)
    if pop_std1 != pop_std2:
        f = (sample_std1**2/pop_std1**2) / (sample_std2**2/pop_std2**2)
    if pop_std1 == pop_std2:
        f = sample_std1**2/sample_std2**2
    return(f)

def bootstrap(DATA,Nbs,num):
    '''
    - DATA is the data that comes in
    - Nbs is the Number of bins
    - num is the number of data points included in each bin 
    - *-*- "num"  needs to equal the sample size -*-*
    '''
    DATA_B = np.empty((Nbs,num))
    for i in range(Nbs):
        DATA_B[i,:] = np.random.choice(DATA,num)
    return(DATA_B)

def temp_timeseries(times,Tmax,title):

    years = np.arange(1980,2015,5)
    fig,ax = plt.subplots(1,1,figsize=(30,10))
    ax.plot(Tmax, c='red', label='Tmax')
    plt.title(title +' Temperature', fontsize=24)
    ax.set_xlabel('Year', fontsize=18)
    ax.set_ylabel('Temperature [$\degree$C]', fontsize=20)
    ax.set_xticks(np.linspace(0,len(times),7))
    ax.set_xticklabels(years, rotation=45)
    ax.set_ylim(-10,50)
    plt.tick_params(labelsize=18)
    plt.legend(fontsize=16, loc='upper left')
    plt.show()
    plt.close()
